\section{Overview}

Scroll Top adds an intuitive icon at the corner of each page so that with one click you will be at the top of the page. No more dragging of the scroll bar. It will be just fun and ergonomic to use.

% ==========================================================
% INSTALLATION
% ==========================================================
\section{Installation}

\subsection{Composer Installation}

\begin{lstlisting}[style=shell]
composer require hk2/scrolltop
\end{lstlisting}

\subsection{Enable Module}

\begin{lstlisting}[style=shell]
bin/magento module:enable HK2_ScrollTop
\end{lstlisting}

\subsection{Run Setup Upgrade}

\begin{lstlisting}[style=shell]
bin/magento setup:upgrade
\end{lstlisting}

\subsection{Compile Dependency Injection}

\begin{lstlisting}[style=shell]
bin/magento setup:di:compile
\end{lstlisting}

\subsection{Deploy Static Content (Production Mode Only)}

\begin{lstlisting}[style=shell]
bin/magento setup:static-content:deploy
\end{lstlisting}

\subsection{Clean Cache}

\begin{lstlisting}[style=shell]
bin/magento cache:clean
\end{lstlisting}

\subsection{Flush Cache}

\begin{lstlisting}[style=shell]
bin/magento cache:flush
\end{lstlisting}

\subsection{Verification}

\begin{enumerate}
    \item Verify the module is enabled.
    \item Verify configuration settings exist.
    \item Verify frontend functionality works.
    \item Verify no errors exist in the Magento logs.
\end{enumerate}

% ==========================================================
% UNINSTALLATION
% ==========================================================
\section{Uninstallation}

\subsection{Disable Module}

\begin{lstlisting}[style=shell]
bin/magento module:disable HK2_ScrollTop
\end{lstlisting}

\subsection{Remove Package}

\begin{lstlisting}[style=shell]
composer remove hk2/scrolltop
\end{lstlisting}

\subsection{Run Setup Upgrade}

\begin{lstlisting}[style=shell]
bin/magento setup:upgrade
\end{lstlisting}

\subsection{Clean and Flush Cache}

\begin{lstlisting}[style=shell]
bin/magento cache:clean && bin/magento cache:flush
\end{lstlisting}

\subsection{Production Mode Cleanup}

\begin{lstlisting}[style=shell]
bin/magento setup:di:compile
bin/magento setup:static-content:deploy
\end{lstlisting}

% ==========================================================
% TROUBLESHOOTING
% ==========================================================
\section{Troubleshooting}

\subsection{Module Not Detected}

\begin{lstlisting}[style=shell]
bin/magento module:status HK2_ScrollTop
\end{lstlisting}

\subsection{Common Installation Issues}

\begin{longtable}{ll}
\toprule
\textbf{Issue} & \textbf{Resolution} \\
\midrule
\endhead
Module not enabled & Verify module status. \\
Composer issue & Verify package installation. \\
Compilation error & Run Dependency Injection compilation again. \\
Cache issue & Flush Magento caches. \\
Permission issue & Verify filesystem ownership and permissions. \\
\bottomrule
\end{longtable}
